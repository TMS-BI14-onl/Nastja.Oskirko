SELECT *
FROM ВОДИТЕЛИ
WHERE СТАЖ NOT IN (3, 5, 6, 10) AND МЕСТО_ПРОПИСКИ='еее'

--OR:

SELECT *
FROM ВОДИТЕЛИ
WHERE СТАЖ!=3 AND 
	СТАЖ!=5 AND
	СТАЖ!=6 AND 
	СТАЖ!=10 AND
	МЕСТО_ПРОПИСКИ='еее'

--Task 2:

SELECT №_ПАСПОРТА
FROM ВОДИТЕЛИ В LEFT JOIN ДИСПЕТЧЕРСКАЯ Д ON В.КОД_ВОДИТЕЛЯ=Д.КОД_ВОДИТЕЛЯ
WHERE Д.КОД_ВОДИТЕЛЯ IS NULL

--Task 3:

SELECT РЕГИСТРАЦИОННЫЙ_№_АВТО, ВРЕМЯ_ПРИБЫТИЯ, ВРЕМЯ_ОТБЫТИЯ, ПУНКТ_НАЗНАЧЕНИЯ
FROM ((ДИСПЕТЧЕРСКАЯ Д LEFT JOIN МАРШРУТ М ON Д.ПУТЕВКА=М.КОД_МАРШРУТА) RIGHT JOIN АВТОМОБИЛИ А ON Д.КОД_АВТО=А.КОД_АВТО)
WHERE А.ГОД_ВЫПУСКА_АВТО=2015
